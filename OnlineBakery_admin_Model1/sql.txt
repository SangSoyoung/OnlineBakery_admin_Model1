/*** 제품 정보 테이블 ***/
create table b_product (
    pcode varchar2(10),
    pname varchar2(30) not null,
    price number(6),
    storage varchar2(30),
    expiration varchar2(30),
    description varchar2(100),
    img varchar2(30),
    constraint pk_code primary key(pcode)
);
desc b_product;
